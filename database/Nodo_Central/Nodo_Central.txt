# SCRIPT — NODO CENTRAL (SQL Server)

CREATE DATABASE nodo_central;
GO

USE nodo_central;
GO

CREATE TABLE cliente (
    id_cliente VARCHAR(20) PRIMARY KEY,

    nombres VARCHAR(100) NOT NULL,
    apellido_paterno VARCHAR(100) NOT NULL,
    apellido_materno VARCHAR(100),

    telefono VARCHAR(20),

    documento_identidad VARCHAR(30) NOT NULL UNIQUE,

    direccion VARCHAR(250),

    email VARCHAR(100),

    fecha_registro DATETIME DEFAULT GETDATE()
);
GO

CREATE TABLE estado (
    id_estado INT PRIMARY KEY,
    descripcion VARCHAR(100) NOT NULL
);
GO

CREATE TABLE almacen (
    id_almacen VARCHAR(20) PRIMARY KEY,

    nombre VARCHAR(100) NOT NULL,

    ciudad VARCHAR(100) NOT NULL,

    direccion VARCHAR(250) NOT NULL,

    nodo_responsable VARCHAR(20) NOT NULL
);
GO


CREATE TABLE ruta (
    id_ruta VARCHAR(20) PRIMARY KEY,

    id_almacen_origen VARCHAR(20) NOT NULL,

    id_almacen_destino VARCHAR(20) NOT NULL,

    distancia DECIMAL(10,2) NOT NULL,

    tiempo_estimado INT NOT NULL,

    CONSTRAINT fk_ruta_origen
        FOREIGN KEY (id_almacen_origen)
        REFERENCES almacen(id_almacen),

    CONSTRAINT fk_ruta_destino
        FOREIGN KEY (id_almacen_destino)
        REFERENCES almacen(id_almacen)
);
GO


CREATE TABLE paquete_global (
    id_paquete VARCHAR(20) PRIMARY KEY,

    codigo_rastreo VARCHAR(50) NOT NULL UNIQUE,

    id_cliente_remitente VARCHAR(20) NOT NULL,
    id_cliente_destinatario VARCHAR(20) NOT NULL,

    id_estado INT NOT NULL,

    id_ruta VARCHAR(20) NOT NULL,

    id_almacen_origen VARCHAR(20) NOT NULL,
    id_almacen_destino VARCHAR(20) NOT NULL,
    id_almacen_actual VARCHAR(20) NOT NULL,

    peso DECIMAL(10,2) NOT NULL,

    volumen DECIMAL(10,2),

    descripcion VARCHAR(300),

    prioridad VARCHAR(20),

    valor_declarado DECIMAL(12,2),

    seguro DECIMAL(12,2),

    costo_envio DECIMAL(12,2),

    fecha_registro DATETIME DEFAULT GETDATE(),

    CONSTRAINT fk_pg_cliente_remitente
        FOREIGN KEY (id_cliente_remitente)
        REFERENCES cliente(id_cliente),

    CONSTRAINT fk_pg_cliente_destinatario
        FOREIGN KEY (id_cliente_destinatario)
        REFERENCES cliente(id_cliente),

    CONSTRAINT fk_pg_estado
        FOREIGN KEY (id_estado)
        REFERENCES estado(id_estado),

    CONSTRAINT fk_pg_ruta
        FOREIGN KEY (id_ruta)
        REFERENCES ruta(id_ruta),

    CONSTRAINT fk_pg_almacen_origen
        FOREIGN KEY (id_almacen_origen)
        REFERENCES almacen(id_almacen),

    CONSTRAINT fk_pg_almacen_destino
        FOREIGN KEY (id_almacen_destino)
        REFERENCES almacen(id_almacen),

    CONSTRAINT fk_pg_almacen_actual
        FOREIGN KEY (id_almacen_actual)
        REFERENCES almacen(id_almacen)
);
GO


CREATE TABLE movimiento_global (
    id_movimiento VARCHAR(20) PRIMARY KEY,

    id_paquete VARCHAR(20) NOT NULL,

    id_almacen VARCHAR(20) NOT NULL,

    fecha_movimiento DATETIME NOT NULL,

    observacion VARCHAR(300),

    CONSTRAINT fk_movglobal_paquete
        FOREIGN KEY (id_paquete)
        REFERENCES paquete_global(id_paquete),

    CONSTRAINT fk_movglobal_almacen
        FOREIGN KEY (id_almacen)
        REFERENCES almacen(id_almacen)
);
GO

CREATE TABLE catalogo_fragmentacion (
    id_fragmento VARCHAR(20) PRIMARY KEY,

    nombre_fragmento VARCHAR(100) NOT NULL,

    tabla_original VARCHAR(100) NOT NULL,

    tipo_fragmentacion VARCHAR(30) NOT NULL,

    nodo_ubicacion VARCHAR(50) NOT NULL,

    descripcion VARCHAR(300)
);
GO


CREATE TABLE log_sincronizacion (
    id_log VARCHAR(20) PRIMARY KEY,

    fecha_sincronizacion DATETIME NOT NULL,

    nodo_origen VARCHAR(20) NOT NULL,

    tabla_afectada VARCHAR(100) NOT NULL,

    operacion VARCHAR(20) NOT NULL,

    cantidad_registros INT NOT NULL,

    estado VARCHAR(20) NOT NULL,

    observacion VARCHAR(300)
);
GO
